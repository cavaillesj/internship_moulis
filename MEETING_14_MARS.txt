\subsection{Adimensionnalisation}
\[
%\lbrace
\begin{array}{rcl}

\end{array}
%\right
\]
With $a\in(0,1)$.

\subsubsection{Equilibrium}
$(N = 0, W = 0) $ stable \\
$(N = 1, W = m) $ stable \\
$(N = a, W = am) $ unstable \\

\section{Fire}

\subsection{definition of the fire}

\paragraph{}

\begin{itemize}
    \item discrete events
    \begin{itemize}
        \item follow a Bernoulli law (frequency $\approx$ 0.01)
    \end{itemize}
    \item strength
    \begin{itemize}
        \item exponential law (average $\approx$ 0.07)
        \item ratio to N and W separately
        \item formally, when a fire occur \\
        \begin{itemize}
            \item ratio $\approx$ exponential\_law 
            \item $N = N - ratio \times N$
            \item $W = W - ratio \times lambda \times N$ 
        \end{itemize}
    \end{itemize}
\end{itemize}

\section{Measures}

\subsection{variability}

\subsubsection{Always}
Compute the variability for all the time and for all the simulation

\subsubsection{Until}
Compute the variability for all simulation but just until it collapse (if the simulation do not collapse, compute for all the time)

\subsubsection{Only}
Compute the variability for all the time but only for simulation that do not collapse

\subsubsection{10\%}
Compute the variability for all simulation but only on the first 10\% time interval. (We assume that most of the simulation have not yet collapse).
